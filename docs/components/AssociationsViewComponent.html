import {
  Component,
  OnInit,
  ElementRef,
  ViewChild,
  Renderer2
} from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { AssociationsService } from '../../services/associations.service';

@Component({
  selector: 'dpdhl-associations-view',
  templateUrl: './associations-view.component.html',
  styleUrls: ['./associations-view.component.scss']
})
export class AssociationsViewComponent implements OnInit {
  constructor(
    private _route: ActivatedRoute,
    private renderer: Renderer2,
    private _router: Router,
    private _associationsService: AssociationsService
  ) {}
  @ViewChild('treeContainer', { read: ElementRef })
  private treeContainer: ElementRef;
  public treeDataSet = [];
  public nodeDetails = {
    id: '',
    node: '',
    node_type: '',
    node_path: ''
  };
  showPop: boolean = false;
  userList: any;
  ngOnInit() {
    this._route.data.subscribe(data => {
      this.treeDataSet = data.resolveAssociationsList.result;
    });
  }

  onclick(eventData, index, event: Event) {
    event.stopPropagation();
    // this.nodeDetails = ;
    this.treeDataSet.forEach(element => {
      if (element.node == eventData.node) {
        this.setNodeData(eventData);
      } else {
        element.children.forEach(elem => {
          if (elem.node == eventData.node) {
            this.setNodeData(eventData);
          } else {
            elem.children.forEach(ele => {
              if (ele.node == eventData.node) {
                this.setNodeData(eventData);
                // if (eventData.node_orig_id == ele.node_orig_id) {
                //   this.userList = this.getUserGroupList();
                //   var index = elem.children.findIndex(
                //     obj => obj.node_orig_id == eventData.node_orig_id
                //   );
                //   console.log(index, this.userList);
                //   if (index > -1 && this.userList && this.userList.length > 0) {
                //     // elem.children.push(this.userList);
                //     console.log(this.userList);
                //   }
                // }
              }
            });
          }
        });
      }
    });
  }

  setNodeData(nodeData) {
    this.showPop = true;
    this.nodeDetails.id = nodeData.node_orig_id;
    this.nodeDetails.node = nodeData.node;
    this.nodeDetails.node_type = nodeData.node_type;
    this.nodeDetails.node_path = nodeData.node_path;
  }

  getUserGroupList() {
    if (this.nodeDetails.node_type == 'User Group') {
      this._associationsService
        .getUserGroups(this.nodeDetails.id)
        .subscribe(res => {
          return res.result;
        });
    }
  }

  editApplication() {
    this._router.navigate([
      '/users/onboard/',
      'Application',
      this.nodeDetails.id
    ]);
  }
}
