import { Component, OnInit, ViewEncapsulation } from '@angular/core';
import { DashboardConstants } from '../../constants/dashboard.constants';
declare let d3: any;
@Component({
  selector: 'dpdhl-user-graph',
  templateUrl: './user-graph.component.html',
  styleUrls: [
    './user-graph.component.scss',
    '../../../../../../node_modules/nvd3/build/nv.d3.css'
  ],
  encapsulation: ViewEncapsulation.None
})
export class UserGraphComponent implements OnInit {
  [x: string]: any;
  data: any;
  options: any;
  calendarValue: any = 2018;
  slidesIndex = 0;
  showCalender = false;
  calendarItem: any = DashboardConstants.YEARCALENDAR;
  graphTitle: string = 'BU Onboarding';
  onBoardingGraphTypes = DashboardConstants.ONBOARDGRAPHTYPE;
  calendarGraphTypes = DashboardConstants.CALENDARGRAPHTYPE;


  constructor() { }
  ngOnInit() {
    this.slidesIndex = this.calendarItem.length - 1;
    this.options = {
      chart: {
        type: 'lineChart',
        height: 450,
       showLegend: false,
        margin: {
          top: 20,
          right: 20,
          bottom: 40,
          left: 55
        },

        useInteractiveGuideline: true,
        dispatch: {},
        xScale: d3.time.scale(),
        xAxis: {
          ticks: d3.time.month,
          axisLabel: 'Month',
          tickFormat: function (d) {
            return d3.time.format('%b')(new Date(d));
          },
        },
        yAxis: {
          axisLabel: 'BU Onboarding',
          axisLabelDistance: -10
        }
      },
      title: {
        enable: true,
        text: 'Title for Line Chart'
      },
      subtitle: {
        enable: true,
        text:
          'Subtitle for simple line chart. Lorem ipsum dolor sit amet, at eam blandit sadipscing, vim adhuc sanctus disputando ex, cu usu affert alienum urbanitas.',
        css: {
          'text-align': 'center',
          margin: '10px 13px 0px 7px'
        }
      },
      caption: {
        enable: false
      }

    };
    // this.chart.legend.margin({ top: 0, right: 0, left: 0, bottom: 0 })
    this.data = DashboardConstants.graphBUData;
  }

  sinAndCos() {
    var sin = [],
      sin2 = [],
      cos = [];

    //Data is represented as an array of {x,y} pairs.
    for (var i = 0; i < 13; i++) {
      sin.push({ x: i, y: Math.sin(i / 10) });
      sin2.push({
        x: i,
        y: i % 10 == 5 ? null : Math.sin(i / 10) * 0.25 + 0.5
      });
      cos.push({ x: i, y: 0.5 * Math.cos(i / 10 + 2) + Math.random() / 10 });
    }

    //Line chart data should be sent as an array of series objects.
    return [
      {
        values: sin, //values - represents the array of {x,y} data points
        key: 'Sine Wave', //key  - the name of the series.
        color: '#ff7f0e' //color - optional: choose your own line color.
      }
    ];
  }

  graphType(value) {
    if (value == 'BU Onboarding') {
      this.data = DashboardConstants.graphBUData;
      this.graphTitle = 'BU Onboarding';
      this.options.chart.yAxis.axisLabel = 'No of BU ';

    }
    if (value == 'User Onboarding') {
      this.data = DashboardConstants.graphUserData;
      this.graphTitle = 'User Onboarding';
      this.options.chart.yAxis.axisLabel = 'No of Users';
    }
    if (value == 'Device Onboarding') {
      this.data = DashboardConstants.graphDeviceData;
      this.graphTitle = 'Device Onboarding';
      this.options.chart.yAxis.axisLabel = 'No of Devices';
    }
    if (value == 'App Onboarding') {
      this.data = DashboardConstants.graphApplicationData;
      this.graphTitle = 'App Onboarding';
      this.options.chart.yAxis.axisLabel = 'No of App';
    }
  }

  calendarType(value) {
    if (value == 'Year') {
      this.options.chart.xAxis.axisLabel = 'Year'
      this.data = DashboardConstants.graphUserData;
      this.calendarItem = DashboardConstants.YEARCALENDAR;
      this.slidesIndex = this.calendarItem.length - 1;
    }
    if (value == 'Month') {
      this.options.chart.xAxis.axisLabel = 'Month'
      var d = new Date();
      this.data = DashboardConstants.graphDeviceData;
      this.calendarItem = DashboardConstants.MONTHCALENDAR;
      let month = this.calendarItem[d.getMonth()];
      this.slidesIndex = this.calendarItem.indexOf(month);
    }
    if (value == 'Week') {
      this.data = DashboardConstants.graphDeviceData;
      this.options.chart.xAxis.axisLabel = 'Week'
      this.calendarItem = DashboardConstants.WEEKCALENDAR;
      this.slidesIndex = 0;
    }
  }

  decreaseCalendar() {
    if (this.slidesIndex > 0) {
      this.slidesIndex--;
    }
  }

  increaseCalendar() {
    if (this.slidesIndex < this.calendarItem.length - 1) {
      this.slidesIndex++;
    }

  }
  openCalender() {
    this.showCalender = !this.showCalender;
  }
  // getWeekNumber(){
  //   var onejan = new Date();
  //   return Math.ceil(((onejan / 86400000) + onejan.getDay()+1)/7);
  // }
}
